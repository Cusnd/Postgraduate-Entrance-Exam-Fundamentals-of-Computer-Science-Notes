\section{数据结构}

\subsection{线性表}
\begin{definition}[线性表]
线性表是具有相同数据类型的 $n$ ($n \geq 0$) 个数据元素的有限序列。
\end{definition}
\begin{note}
\begin{itemize}
    \item \textbf{顺序表}：随机存取，存储密度高，但插入删除需要移动元素。插入平均移动 $n/2$，删除平均移动 $(n-1)/2$。
    \item \textbf{链表}：不支持随机存取，插入删除方便（修改指针）。头指针指向第一个结点，头结点方便操作。
\end{itemize}
\end{note}

\subsection{栈、队列和数组}
\begin{itemize}
    \item \textbf{栈 (Stack)}: 后进先出 (LIFO)。$n$ 个不同元素进栈，出栈序列个数为 $\frac{1}{n+1}C_{2n}^n$ (卡特兰数)。
    \item \textbf{队列 (Queue)}: 先进先出 (FIFO)。循环队列判满条件：$(rear + 1) \% MaxSize == front$。
    \item \textbf{特殊矩阵}：对称矩阵、三角矩阵、三对角矩阵的压缩存储（下标映射公式）。
\end{itemize}

\subsection{树与二叉树}
\begin{theorem}[二叉树性质]
\begin{enumerate}
    \item 非空二叉树上的叶子结点数等于度为2的结点数加1，即 $n_0 = n_2 + 1$。
    \item 二叉树第 $i$ 层至多有 $2^{i-1}$ 个结点。
    \item 高度为 $h$ 的二叉树至多有 $2^h - 1$ 个结点。
    \item 具有 $n$ 个结点的完全二叉树的高度为 $\lceil \log_2(n+1) \rceil$ 或 $\lfloor \log_2 n \rfloor + 1$。
\end{enumerate}
\end{theorem}
\begin{itemize}
    \item \textbf{遍历}: 前序 (Root-L-R), 中序 (L-Root-R), 后序 (L-R-Root), 层序。
    \item \textbf{线索二叉树}: 利用空指针域存放前驱/后继。$n$ 个结点的二叉树有 $n+1$ 个空链域。
    \item \textbf{树与森林}: 树的存储（双亲表示法、孩子兄弟表示法）。树、森林与二叉树的转换（左孩子右兄弟）。
    \item \textbf{哈夫曼树}: 带权路径长度 (WPL) 最小的二叉树。构造：每次选权值最小的两棵树合并。
\end{itemize}

\subsection{图}
\begin{itemize}
    \item \textbf{存储}: 邻接矩阵 (适合稠密图, 唯一), 邻接表 (适合稀疏图, 不唯一)。
    \item \textbf{遍历}: BFS (广度优先, 需辅助队列, 类似层序), DFS (深度优先, 需各种/栈, 类似先序)。
    \item \textbf{最小生成树 (MST)}:
    \begin{itemize}
         \item Prim算法: 关注点，适合稠密图，复杂度 $O(|V|^2)$。
         \item Kruskal算法: 关注边，适合稀疏图，复杂度 $O(|E|\log|E|)$。
    \end{itemize}
    \item \textbf{最短路径}:
    \begin{itemize}
         \item Dijkstra: 单源最短路，不适用负权边，$O(|V|^2)$。
         \item Floyd: 多源最短路，适用负权（但在负环时不适用），$O(|V|^3)$。
    \end{itemize}
    \item \textbf{拓扑排序}: AOV网，检测有向图是否有环。
    \item \textbf{关键路径}: AOE网，从源点到汇点的最长路径，决定工程工期。
\end{itemize}

\subsection{查找}
\begin{itemize}
    \item \textbf{折半查找}: 仅适用于有序顺序表。判定树为平衡二叉树，高度 $\lceil \log_2(n+1) \rceil$，平均查找长度 $O(\log n)$。
    \item \textbf{分块查找}: 块内无序，块间有序。
    \item \textbf{B树}: 平衡的多路查找树。$m$ 阶B树所有叶结点在同一层，关键字个数 $\lceil m/2 \rceil - 1 \leq n \leq m-1$。
    \item \textbf{B+树}: 适合文件系统/数据库索引。所有关键字都在叶子结点出现。
    \item \textbf{散列表}: 散列函数（除留余数法），冲突处理（开放定址法、拉链法）。装填因子 $\alpha = n/m$。
    \item \textbf{KMP算法}: $next$ 数组的计算，利用最大相等前后缀。
\end{itemize}

\section{计算机组成原理}

\subsection{计算机系统概述}
\begin{itemize}
    \item \textbf{冯·诺依曼结构}: "存储程序"概念。由于指令和数据都存放在内存中，形式上由于没有区别，必须依据指令周期的不同阶段来区分。
    \item \textbf{性能指标}:
    \begin{itemize}
        \item CPU主频 ($f$)、时钟周期 ($T=1/f$)、CPI (执行一条指令所需的时钟周期数)。
        \item MIPS (每秒百万条指令) = $f / (CPI \times 10^6)$。
        \item CPU执行时间 = (指令条数 $\times$ CPI) / 主频。
    \end{itemize}
\end{itemize}

\subsection{数据的表示和运算}
\begin{itemize}
    \item \textbf{补码}: $n+1$ 位补码范围 $[-2^n, 2^n-1]$。0有唯一表示。
    \item \textbf{IEEE 754浮点数}:
    \begin{itemize}
        \item 格式: 符号位(S) + 阶码(E, 移码) + 尾数(M, 原码)。
        \item float (32位): 1+8+23, 偏置值 127。
        \item double (64位): 1+11+52, 偏置值 1023。
    \end{itemize}
    \item \textbf{大端/小端}: 大端(Big-Endian)高字节存低地址；小端(Little-Endian)低字节存低地址。
\end{itemize}

\subsection{存储系统}
\begin{itemize}
    \item \textbf{层次结构}: 寄存器 -> Cache -> 主存 -> 辅存。解决速度、容量、价格矛盾。
    \item \textbf{SRAM vs DRAM}: SRAM (触发器, 快, 贵, Cache); DRAM (电容, 慢, 便宜, 需刷新, 主存)。
    \item \textbf{Cache映射}:
    \begin{itemize}
        \item 直接映射: $j = i \mod 2^c$。冲突率高。
        \item 全相联映射: 任意位置。比较器复杂。
        \item 组相联映射: $j = i \mod Q$。折中方案。
    \end{itemize}
    \item \textbf{虚拟存储器}: 逻辑地址 -> 物理地址。页式管理 (Page), 段式管理 (Segment)。TLB (快表) 加速地址变换。
\end{itemize}

\subsection{指令系统}
\begin{itemize}
    \item \textbf{寻址方式}: 立即寻址、寄存器寻址、直接寻址、间接寻址、寄存器间接寻址、变址寻址(数组)、相对寻址(跳转)、基址寻址(重定位)。
    \item \textbf{CISC vs RISC}: CISC (指令丰富, 变长, 微程序); RISC (常用指令, 定长, 硬布线, 流水线)。
\end{itemize}

\subsection{中央处理器}
\begin{itemize}
    \item \textbf{控制器}: 硬布线 (速度快, 修改难), 微程序 (由微指令组成微程序, 存储在控制存储器CM中)。
    \item \textbf{指令流水线}: IF, ID, EX, MEM, WB。
    \item \textbf{流水线冒险}:
    \begin{itemize}
        \item 结构冒险 (资源冲突): 多端口存储器, 指令/数据Cache分离。
        \item 数据冒险 (数据依赖): 前推/旁路 (Forwarding), 暂停 (Stall)。
        \item 控制冒险 (跳转): 分支预测, 延迟槽。
    \end{itemize}
\end{itemize}

\subsection{总线}
\begin{itemize}
    \item \textbf{带宽}: $B = W \times f / N$ (宽度 $\times$ 频率 / 周期数)。
    \item \textbf{仲裁}: 链式查询 (对电路故障敏感), 计数器查询, 独立请求。
\end{itemize}

\subsection{输入/输出系统}
\begin{itemize}
    \item \textbf{I/O方式}:
    \begin{itemize}
        \item 程序查询: CPU忙等。
        \item 中断: 异常或外部请求。CPU响应中断条件 (指令执行结束, 开中断)。
        \item DMA: 直接存储器访问。仅在传送开始和结束时需CPU干预。块传输。
    \end{itemize}
\end{itemize}

\section{操作系统}

\subsection{操作系统概述}
\begin{itemize}
    \item \textbf{特征}: 并发 (Concurrency)、共享 (Sharing)、虚拟 (Virtual)、异步 (Asynchrony)。
    \item \textbf{运行环境}:
    \begin{itemize}
        \item 内核态 (Kernel Mode): 执行特权指令。
        \item 用户态 (User Mode): 执行非特权指令。
        \item 中断 (Interrupt): 外中断 (I/O) 和 内中断 (异常/陷阱)。
    \end{itemize}
\end{itemize}

\subsection{进程管理}
\begin{itemize}
    \item \textbf{状态转换}: 就绪 (Ready) -> 运行 (Running) -> 阻塞 (Blocked)。
    \item \textbf{调度算法}: FCFS, SJF, 优先级, 时间片轮转, 多级反馈队列。
    \item \textbf{同步与互斥}: 信号量 (PV操作)。
    \begin{itemize}
        \item P操作 (wait): S--, 若S<0则阻塞。
        \item V操作 (signal): S++, 若S<=0则唤醒。
    \end{itemize}
    \item \textbf{死锁}:
    \begin{itemize}
        \item 4个必要条件: 互斥、请求与保持、不剥夺、循环等待。
        \item 处理: 预防 (破坏条件), 避免 (银行家算法), 检测与解除。
    \end{itemize}
\end{itemize}

\subsection{内存管理}
\begin{itemize}
    \item \textbf{分页}: 逻辑上连续，物理上离散。页表。不会产生外部碎片，有内部碎片。
    \item \textbf{分段}: 逻辑分段。有外部碎片。
    \item \textbf{段页式}: 先分段，段内分页。
    \item \textbf{虚拟内存}: 局部性原理。TLB, Page Fault。
    \item \textbf{置换算法}: OPT (最佳, 不可实现), FIFO (Belady异常), LRU (最近最久未使用), CLOCK (时钟置换).
\end{itemize}

\subsection{文件管理}
\begin{itemize}
    \item \textbf{FCB}: 文件控制块。
    \item \textbf{索引结点 (Inode)}: 将文件名与文件描述信息分开，减少目录项大小，提高检索速度。
    \item \textbf{分配方式}: 连续分配 (快, 碎片), 链接分配 (隐式/显式FAT), 索引分配 (支持大文件)。
\end{itemize}

\subsection{输入/输出 (I/O) 管理}
\begin{itemize}
    \item \textbf{SPOOLing技术}: 假脱机。将独占设备改造为共享设备。
    \item \textbf{缓冲区}: 单缓冲、双缓冲、循环缓冲、缓冲池。
\end{itemize}

\section{计算机网络}

\subsection{计算机网络体系结构}
\begin{itemize}
    \item \textbf{OSI 7层}: 物、链、网、输、会、表、应。
    \item \textbf{TCP/IP 4层}: 网络接口层、网际层 (IP)、传输层 (TCP/UDP)、应用层。
    \item \textbf{性能指标}: 带宽、吞吐量、时延 (发送+传播+处理+排队)、时延带宽积、RTT。
\end{itemize}

\subsection{物理层}
\begin{itemize}
    \item \textbf{奈奎斯特定理}: 无噪声，$C = 2W \log_2 V$。
    \item \textbf{香农定理}: 有噪声，$C = W \log_2 (1+S/N)$。
    \item \textbf{编码}: 曼彻斯特 (中间跳变, 自带时钟), 差分曼彻斯特。
    \item \textbf{设备}: 中继器 (集线器Hub) - 物理层，不能隔离冲突域/广播域。
\end{itemize}

\subsection{数据链路层}
\begin{itemize}
    \item \textbf{功能}: 封装成帧, 透明传输, 差错控制 (CRC, FCS)。
    \item \textbf{流量控制}:
    \begin{itemize}
        \item 停止-等待: $W_T=1, W_R=1$。
        \item GBN (后退N帧): $W_T > 1, W_R=1$。
        \item SR (选择重传): $W_T > 1, W_R > 1$。
    \end{itemize}
    \item \textbf{介质访问}: CSMA/CD (以太网, 先听后发, 边听边发, 冲突停止, 二进制指数退避)。
    \item \textbf{设备}: 网桥 (交换机Switch) - 链路层，隔离冲突域，不隔离广播域。
\end{itemize}

\subsection{网络层}
\begin{itemize}
    \item \textbf{IP协议}:
    \begin{itemize}
        \item 头部: 至少20字节。TTL (生存时间)。
        \item NAT: 私有IP转公网IP。
    \end{itemize}
    \item \textbf{子网划分}: CIDR (无分类编址), 子网掩码。
    \item \textbf{协议}: ARP (IP->MAC), DHCP (自动分配IP), ICMP (Ping, Traceroute)。
    \item \textbf{路由协议}:
    \begin{itemize}
        \item RIP: 距离向量, 跳数限制15, UDP。
        \item OSPF: 链路状态, Dijkstra算法, IP。
        \item BGP: 路径向量, TCP。
    \end{itemize}
    \item \textbf{设备}: 路由器 (Router) - 网络层，隔离冲突域，隔离广播域。
\end{itemize}

\subsection{传输层}
\begin{itemize}
    \item \textbf{TCP}: 面向连接, 可靠, 字节流。
    \begin{itemize}
        \item 连接: 三次握手 (SYN, SYN+ACK, ACK), 四次挥手。
        \item 拥塞控制: 慢开始 (exp), 拥塞避免 (linear), 快重传 (3 ACK), 快恢复。
    \end{itemize}
    \item \textbf{UDP}: 无连接, 不可靠, 报文段。
\end{itemize}

\subsection{应用层}
\begin{itemize}
    \item \textbf{DNS}: 域名解析, UDP 53。递归/迭代查询。
    \item \textbf{FTP}: 文件传输, 控制连接 21 (一直开), 数据连接 20 (主动) / 随机 (被动)。
    \item \textbf{Email}: SMTP (推, TCP 25), POP3 (拉, TCP 110), IMAP。
    \item \textbf{HTTP}: 80。1.0 (非持久), 1.1 (持久)。
\end{itemize}
